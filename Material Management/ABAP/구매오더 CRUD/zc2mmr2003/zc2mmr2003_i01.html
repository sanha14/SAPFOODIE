MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'SEARCH'.
      CLEAR gv_okcode.
      PERFORM get_data.
      PERFORM refresh_grid.

    WHEN 'CONFIRM'.
      CLEAR gv_okcode.
      PERFORM confirm_data.
      PERFORM refresh_grid.
      PERFORM refresh_grid2.

    WHEN 'REFRESH'.
      CLEAR gv_okcode.
      PERFORM set_data.
      PERFORM refresh_grid.
      PERFORM refresh_grid2.
  ENDCASE.
ENDMODULE.
