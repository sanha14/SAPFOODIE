FORM set_radio .

  LOOP AT SCREEN.
    CASE 'X'.
      WHEN pa_rd1.
        title = '매입전표 생성'.
        IF screen-group1 = 'SC2'.
           screen-active = 1.
        endif.
        if screen-group1 = 'SC1'.
          screen-active = 0.
        ENDIF.

      WHEN pa_rd2.
       title = '매출전표 생성'.
        IF screen-group1 = 'SC1'.
          screen-active = 1.
        endif.
        if screen-group1 = 'SC2'.
          screen-active = 0.
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.
  ENDLOOP.
ENDFORM.
